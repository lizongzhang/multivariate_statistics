library(tidyverse)
library(DescTools)
decathlon <- decathlon %>% mutate(tier = case_when(
  rank %[]% c(1,33) ~ "first",
  rank %[]% c(34,66) ~ "second",
  rank %[]% c(67,100) ~ "third",
  ))
library(MASS)
model <- lda(tier ~ M100 + Hurdles_M100 + M400 + M1500 +
             long_jump + high_jump + pole_vault +
             shot_put + discus_thow + javelin_throw, decathlon)
