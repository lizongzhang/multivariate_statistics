mpg %>% 
  filter(year == 1999) %>% 
  select(displ,cty, hwy,trans) %>% 
  arrange(displ) %>% 
  mutate(cty.kpl = 0.425*cty, 
               hwy.kpl = 0.425*hwy) %>% 
  mutate(transmission = 
           if_else(substring(trans, 1,4) == "auto", 
                   "auto","manual"))
